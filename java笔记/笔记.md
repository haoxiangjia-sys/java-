// 1. 定义“类” (Class)
class GamePlayer {
    // --- 属性 (Attributes) ---
    String name;
    int health;

    // --- 构造函数 (Constructor) ---
    // 名字必须和类名完全一致，没有返回类型
    // 当你 "new" 的时候，它会被自动调用
    public GamePlayer(String inputName, int inputHealth) {
        System.out.println("正在根据图纸生产玩家对象...");
        this.name = inputName;    // 初始化属性
        this.health = inputHealth;
    }

    // --- 方法 (Methods) ---
    void takeDamage(int amount) {
        this.health -= amount;
        System.out.println(name + " 失去了 " + amount + " 点血量。");
    }
}

public class Main {
    public static void main(String[] args) {
        // 2. 实例化 (Instantiation)
        // 使用 new 关键字，触发构造函数
        GamePlayer player1 = new GamePlayer("阿尔法狗", 100); 
        GamePlayer player2 = new GamePlayer("柯洁", 90);

        // 3. 操作对象
        player1.takeDamage(20); // 调用方法
        System.out.println(player2.name + " 的当前血量: " + player2.health); // 访问属性
    }
}
